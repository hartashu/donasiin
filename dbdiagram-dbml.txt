Table users {
  _id ObjectId [pk]
  avatarUrl String
  username String [unique]
  fullName String
  email String [unique]
  password String
  address String
  location Object [note: 'GeoJSON: { type: "Point", coordinates: [lng, lat] }']
  dailyLimit Number
  createdAt Date
  updatedAt Date
}

Table posts {
  _id ObjectId [pk]
  title String
  slug String [unique]
  thumbnailUrl String
  imageUrls Array
  description String
  category String
  isAvailable Boolean
  userId ObjectId [ref: > users._id]
  aiAnalysis String
  carbonKg Number
  createdAt Date
  updatedAt Date
}

Table requests {
  _id ObjectId [pk]
  userId ObjectId [ref: > users._id]
  postId ObjectId [ref: > posts._id]
  status String [note: 'Enum: PENDING, ACCEPTED, REJECTED, SHIPPED, COMPLETED']
  trackingCode String
  createdAt Date
  updatedAt Date
}

Table messages {
  _id ObjectId [pk]
  conversationId String
  senderId ObjectId [ref: > users._id]
  receiverId ObjectId [ref: > users._id]
  text String
  createdAt Date
}
